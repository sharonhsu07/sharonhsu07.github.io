Automated Localization Tool

一鍵替換 PSD 中的所有對話與字體樣式。此腳本專為多語言漫畫發行設計，能大幅縮短繁瑣的排版時間。
透過讀取定義好的 CSV 設定檔，程式會自動識別角色台詞，並套用對應的語言專用字體與樣式。節省繁瑣的複製貼上步驟，讓設計師直接進到調整設計階段。
* 支援批量處理文件夾內的所有 PSD 檔案。 自動偵測圖層結構，無需手動選取文字層。

(初始化代碼省略) } // === 核心修正處：parseCSVLines === function parseCSVLines(text) { var lines = []; var currentRow = []; var currentCell = ""; var insideQuote = false; for (var i = 0; i < text.length; i++) { var c = text[i]; var nextC = (i + 1 < text.length) ? text[i + 1] : ""; if (c === '"') { if (insideQuote && nextC === '"') { currentCell += '"'; i++; } else { insideQuote = !insideQuote; } } else if (c === ',' && !insideQuote) { currentRow.push(currentCell); currentCell = ""; } else if ((c === '\r' || c === '\n') && !insideQuote) { if (c === '\r' && nextC === '\n') i++; currentRow.push(currentCell); lines.push(currentRow); currentRow = []; currentCell = ""; } else { currentCell += c; } } // 補上最後一行 if (currentCell !== "" || currentRow.length > 0) { currentRow.push(currentCell); lines.push(currentRow); } return lines; }
